\subsection{Constraint on foot positions}

\begin{equation*}
    \diag{M}{\begin{bmatrix} a & b \end{bmatrix}}
    \FP
    \le
    \M{1} c
\end{equation*}

\begin{equation*}
    \diag{M}{\begin{bmatrix} a & b \end{bmatrix}}
    \left(
        \M{I}_{fp0}
        \fp_0
        +
        \M{I}_{\FD}
        \FD
    \right)
    \le
    \M{1} c
\end{equation*}

\begin{equation*}
    \diag{M}{\begin{bmatrix} a & b \end{bmatrix}}
    \M{I}_{\FD}
    \FD
    \le
    \M{1} c -
    \diag{M}{\begin{bmatrix} a & b \end{bmatrix}}
    \M{I}_{fp0}
    \fp_0
\end{equation*}


\begin{equation*}
    \begin{bmatrix}
        \M{0} & \diag{M}{\begin{bmatrix} a & b \end{bmatrix}} \M{I}_{\FD}
    \end{bmatrix}
    \M{X}
    \le
    \M{1} c -
    \diag{M}{\begin{bmatrix} a & b \end{bmatrix}}
    \M{I}_{fp0}
    \fp_0
\end{equation*}
